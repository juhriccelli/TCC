
SELECT candidatos.* , eleicoes.sum(qt_votos_nominais)
FROM candidatos, eleicoes
WHERE nm_candidato LIKE "%shéridan%"
AND candidatos.ano_eleicao = eleicoes.ano_eleicao
GROUP BY ano_eleicao
